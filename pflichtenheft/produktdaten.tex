\section{Produktdaten}

\subsection{Profildaten}
\begin{requirements}
	\req{D10} Über ein Spielerprofil sind folgende Daten zu speichern:\\
	Spielername, Profilbild (optional)
	\req{D20} Über den Spielfortschritt sind folgende Daten zu speichern:\\
	Levelfortschritt/erreichtes Level, erworbene Achievements, Achievementfortschritt, statistische Daten
\end{requirements}

\subsection{Einstellungen}
\begin{requirements}
	\req{D30} Folgende Einstellungen sind zu speichern:\\
	Zoom (Multitouch/Buttons), Soundoptionen: Soundtrack \& Effekte (an/aus, Lautstärke)
\end{requirements}

\subsection{Optionale Daten}
\begin{requirements}
	\req{D40} Falls eine Option zum Erstellen von Leveln im Sandbox-Modus zur Verfügung steht, müssen folgende Daten gespeichert werden:\\
	Erstellte Level
\end{requirements}
